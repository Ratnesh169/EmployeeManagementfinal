CREATE TABLE Employee (
    EmployeeCode INT,
    EmployeeName VARCHAR(50),
    DateOfBirth DATETIME,
    Gender BIT,
    Department VARCHAR(20),
    Designation VARCHAR(20),
    BasicSalary FLOAT
);
